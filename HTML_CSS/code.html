public static double CalcArea(double a, double b, double c)
{
if (Operation.ExamTriangle(a, b, c) == true)
{
double p = (a + b + c) / 2;
double S = Math.Sqrt(p * (p - a) * (p - b) * (p - c));
return S;
}
else
{
double S = -1;
Console.WriteLine("Введены не верные значения сторон треугольника");
return S;
}
}
